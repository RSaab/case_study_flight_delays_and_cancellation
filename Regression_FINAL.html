Primary thoughts on the observed model¶

The linear regression model shows a significant value for the R-squared statistic and the model's p-value is significant to reject the null hypothesis indicating that the coefficients are not all equal to zero.

On the other hand, adding polynomial interactions significantly increased the R-squared metric but the adjusted R-squared shows little improvement. Also, the log-likelihood shows a small increase as well. This indicates that we may be overfitting.

To verify the model we will run some tests on the assumptions of a linear regression model.

def homoscedasticity_test(model):
    fitted_vals = model.predict()
    resids = model.resid
    resids_standardized = model.get_influence().resid_studentized_internal

    fig, ax = plt.subplots(1,2)

    sns.regplot(x=fitted_vals, y=resids, lowess=True, ax=ax[0], line_kws={'color': 'red'})
    ax[0].set_title('Residuals vs Fitted', fontsize=16)
    ax[0].set(xlabel='Fitted Values', ylabel='Residuals')

    sns.regplot(x=fitted_vals, y=np.sqrt(np.abs(resids_standardized)), lowess=True, ax=ax[1], line_kws={'color': 'red'})
    ax[1].set_title('Scale-Location', fontsize=16)
    ax[1].set(xlabel='Fitted Values', ylabel='sqrt(abs(Residuals))')

    bp_test = pd.DataFrame(sms.het_breuschpagan(resids, model.model.exog), 
                           columns=['value'],
                           index=['Lagrange multiplier statistic', 'p-value', 'f-value', 'f p-value'])

    gq_test = pd.DataFrame(sms.het_goldfeldquandt(resids, model.model.exog)[:-1],
                           columns=['value'],
                           index=['F statistic', 'p-value'])

    print(' Breusch-Pagan test ----')
    print(bp_test)
    print('\n Goldfeld-Quandt test ----')
    print(gq_test)
    print('\n Residuals plots ----')
    plt.show()

  • The Breusch-Pagan tests the null hypothesis that all variances are equal and the given p-value is significant enough at a critical value of 0.05 to reject it. Therefore variances are not equal.
  • The Goldfeld-Quandt test tests the null hypothesis that the variances of the errors are not constant and the p-value is not significant to reject it at a critical level of 0.05. Therefore the varaince of the errors is probably not constant.
  • The above tests indicate homoscedasticity and thus violating the assumption of linear regression.

def normality_of_residuals_test(model):
    sm.ProbPlot(model.resid).qqplot(line='s');
    plt.title('Q-Q plot');

    jb = stats.jarque_bera(model.resid)
    sw = stats.shapiro(model.resid)
    ad = stats.anderson(model.resid, dist='norm')
    ks = stats.kstest(model.resid, 'norm')

    print(f'Shapiro-Wilk test ---- statistic: {sw[0]:.4f}, p-value: {sw[1]:.4f}')
    print(f'Kolmogorov-Smirnov test ---- statistic: {ks.statistic:.4f}, p-value: {ks.pvalue:.4f}')

As a conclusion, even after taking the log transformation of the dependent variable, the above violations indicate that we cannot use linear regression and hence we will look at observing other models.

# print top n features
def print_features_with_rank(features, importances):
    zipped_and_sorted = sorted(zip(features, importances),reverse = True, key=lambda x: x[1])
    str = ''
    for x in zipped_and_sorted:
        str += (f"|{x[0]}: {x[1]:.5}")
    print(str+ "|")

def get_top_features(features, importances):
    zipped_and_sorted = sorted(zip(features, importances), key=lambda x: x[1])
    result = [x[0] for x in zipped_and_sorted if x[1] == 1]
    return result

def run_rfe_and_score_best_feature(model, x_train, y_train, x_test, y_test):
    print('--------------------------------------------------------------')
    print(f"\033[1m {type(model).__name__} \033[0m: ", end = '')
    selector = RFE(model, 10, step=1)
    selector = selector.fit(x_train,y_train)

    features = get_top_features(x_train, selector.ranking_)
    model.fit(x_train[features],y_train)
    
    preds=model.predict(x_test[features])
    print(f"\033[1m R squared:\033[0m {r2_score(y_test,preds):.5} | \033[1m RMSE: \033[0m{sqrt(mean_squared_error(y_test, preds)):.5}",)
    print_features_with_rank(features, model.feature_importances_)

def find_relevant_variable(component, threshold, data):
    variables = []
    weights = []
    
    for i, var in zip(range(0,len(data.keys())), data.keys()):
        if abs(component[i])>threshold:
            variables.append(var)
            weights.append(component[i])
    
    return variables, weights

    
def run_pca(pca_data, n, plot=True):
    X_pca = normalize(pca_data)

    if n == None:
        pca = PCA()
    else:
        pca = PCA(n_components=n)

    pca.fit(X_pca)

    if plot:
        plt.plot(range(1,len(pca.components_)+1),pca.explained_variance_ratio_,'-o')
        plt.xlabel('components')
        plt.ylabel('% explained variance')
        plt.title("Scree plot")
        plt.show()
    
    return pca
